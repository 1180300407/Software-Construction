package Exceptions;

public class InconsisitentPlaneCompositionException extends IncorrectElementDependencyException{

	/**
	 * 
	 */
	private static final long serialVersionUID = 4059757491601624096L;
	
	public InconsisitentPlaneCompositionException(String errorMessage) {
		super(errorMessage);
	}
	
	public InconsisitentPlaneCompositionException() {
		super();
	}
	
	@Override
	public String getErrorMessage() {
		return "飞机编号相同,但其他部分不同!"+super.getErrorMessage();
	}
}
